BETWEEN __ AND: (USA-SE SOMENTE COM NUMERO)

SELECT ALU.COD_ALUNO "CODIGO", ALU.NOME_ALUNO "ALUNO" , DISCI.NOME_DISC "DISCI", NOTA.NOTA_ALUNO "NOTA"
FROM TB_ALUNOS ALU 
INNER JOIN TB_NOTAS NOTA
ON ALU.COD_ALUNO = NOTA.COD_NOTA_ALUNO
INNER JOIN TB_DISCIPLINAS DISCI
ON DISCI.COD_DISC = NOTA.COD_NOTA_DISC
WHERE NOTA.NOTA_ALUNO BETWEEN 7 AND 9

SE QUISER QUE O PRIMEIRO CARACTER SEJA 'K' E O RESTO TANTO FAZ:

SELECT ALU.COD_ALUNO "CODIGO", ALU.NOME_ALUNO "ALUNO" , DISCI.NOME_DISC "DISCI", NOTA.NOTA_ALUNO "NOTA"
FROM TB_ALUNOS ALU 
LEFT JOIN TB_NOTAS NOTA
ON ALU.COD_ALUNO = NOTA.COD_NOTA_ALUNO
LEFT JOIN TB_DISCIPLINAS DISCI
ON DISCI.COD_DISC = NOTA.COD_NOTA_DISC
WHERE ALU.NOME_ALUNO LIKE 'K%'
