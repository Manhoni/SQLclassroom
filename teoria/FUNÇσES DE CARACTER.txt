FUNÇOES DE CARACTER

LOWER

SELECT ALU.COD_ALUNO "CODIGO", LOWER (ALU.NOME_ALUNO) "ALUNO" , DISCI.NOME_DISC "DISCI", NOTA.NOTA_ALUNO "NOTA"
FROM TB_ALUNOS ALU 
LEFT JOIN TB_NOTAS NOTA
ON ALU.COD_ALUNO = NOTA.COD_NOTA_ALUNO
LEFT JOIN TB_DISCIPLINAS DISCI
ON DISCI.COD_DISC = NOTA.COD_NOTA_DISC
