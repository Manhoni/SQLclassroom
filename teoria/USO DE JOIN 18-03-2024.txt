USO DE JOIN

EXEMPLOS 

O ASTERISCO NO SELECT SIGNIFICA QUE VAI PUXAR TODAS AS COLUNAS DA TABELA.

INNER JOIN:

SELECT * 
FROM TB_ALUNOS 
INNER JOIN TB_NOTAS
ON TB_ALUNOS.COD_ALUNO = TB_NOTAS.COD_NOTA_ALUNO 

LEFT JOIN:

SELECT * 
FROM TB_ALUNOS 
LEFT JOIN TB_NOTAS
ON TB_ALUNOS.COD_ALUNO = TB_NOTAS.COD_NOTA_ALUNO 

RIGHT JOIN:

SELECT * 
FROM TB_ALUNOS 
RIGHT JOIN TB_NOTAS
ON TB_ALUNOS.COD_ALUNO = TB_NOTAS.COD_NOTA_ALUNO

FULL OUTER JOIN:

SELECT * 
FROM TB_ALUNOS 
FULL OUTER JOIN TB_NOTAS
ON TB_ALUNOS.COD_ALUNO = TB_NOTAS.COD_NOTA_ALUNO

CROSS JOIN:

SELECT * 
FROM TB_ALUNOS 
CROSS JOIN TB_NOTAS


PUXAR SOMENTE ALGUMAS COLUNAS DA TABELA:

SELECT COD_ALUNO, NOME_ALUNO , NOTA_ALUNO
FROM TB_ALUNOS 
INNER JOIN TB_NOTAS
ON TB_ALUNOS.COD_ALUNO = TB_NOTAS.COD_NOTA_ALUNO



USO DE ALIAS PARA SIMPLICAR NOME DE TABELAS: (ALIAS S√ÉO APELIDOS QUE DAMOS PARA TABELAS OU COLUNAS)

SELECT ALU.COD_ALUNO, ALU.NOME_ALUNO , NOTA.NOTA_ALUNO
FROM TB_ALUNOS ALU 
LEFT JOIN TB_NOTAS NOTA
ON ALU.COD_ALUNO = NOTA.COD_NOTA_ALUNO


USO DE ALIAS PARA SIMPLICAR NOME DE COLUNAS:

SELECT ALU.COD_ALUNO "CODIGO", ALU.NOME_ALUNO "ALUNO" , NOTA.NOTA_ALUNO "NOTA"
FROM TB_ALUNOS ALU 
LEFT JOIN TB_NOTAS NOTA
ON ALU.COD_ALUNO = NOTA.COD_NOTA_ALUNO
