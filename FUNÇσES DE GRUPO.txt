FUNÇÕES DE 


FUNÇÃO AVG

MÉDIA:

SELECT AVG(NOTA_ALUNO) FROM TB_NOTAS;

MÉDIA POR ALUNO:

SELECT 
       TB_NOTAS.COD_NOTA_ALUNO             COD_ALUNO, 
       TB_ALUNOS.NOME_ALUNO,
       ROUND(AVG(NOTA_ALUNO),2)AS MEDIAS
FROM TB_NOTAS
INNER JOIN TB_ALUNOS
ON TB_NOTAS.COD_NOTA_ALUNO=TB_ALUNOS.COD_ALUNO
GROUP BY COD_NOTA_ALUNO, NOME_ALUNO
ORDER BY COD_ALUNO;










FUNÇÃO COUNT

COM USO DO * NO COUNT - SE USAR RIGHT VAI CONTAR 1 NOS NULOS

SELECT 
       TB_NOTAS.COD_NOTA_ALUNO    COD_ALUNO, 
       TB_ALUNOS.NOME_ALUNO,
       ROUND(AVG(NOTA_ALUNO),2)AS MEDIAS,
       COUNT(*)                   NUM_NOTAS
FROM TB_NOTAS
INNER JOIN TB_ALUNOS
ON TB_NOTAS.COD_NOTA_ALUNO=TB_ALUNOS.COD_ALUNO
GROUP BY COD_NOTA_ALUNO, NOME_ALUNO
ORDER BY COD_ALUNO;

COM O USO DO COD_NOTA_ALUNO - ZERA OS NULOS

SELECT 
       TB_NOTAS.COD_NOTA_ALUNO    COD_ALUNO, 
       TB_ALUNOS.NOME_ALUNO,
       ROUND(AVG(NOTA_ALUNO),2)AS MEDIAS,
       COUNT(COD_NOTA_ALUNO)                   NUM_NOTAS
FROM TB_NOTAS
INNER JOIN TB_ALUNOS
ON TB_NOTAS.COD_NOTA_ALUNO=TB_ALUNOS.COD_ALUNO
GROUP BY COD_NOTA_ALUNO, NOME_ALUNO
ORDER BY COD_ALUNO;











USO DE HAVING

SELECT 
       COD_ALUNO, 
       TB_ALUNOS.NOME_ALUNO,
       ROUND(AVG(NOTA_ALUNO),2)AS MEDIAS,
       COUNT(COD_NOTA_ALUNO)                   NUM_NOTAS
FROM TB_NOTAS
INNER JOIN TB_ALUNOS
ON TB_NOTAS.COD_NOTA_ALUNO=TB_ALUNOS.COD_ALUNO
GROUP BY COD_ALUNO, NOME_ALUNO
HAVING COUNT(COD_NOTA_ALUNO)>1
ORDER BY COD_ALUNO;
